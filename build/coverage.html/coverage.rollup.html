function GetActionOp(op, expand, minargs) {
  minargs = minargs || 1;

        items.push([op].concat(pin, values[i]));
      }
    } else {
      items.push([op].concat(values));
    }
    namespace.set('ops', items);
  };
  return ActionOp;
};
